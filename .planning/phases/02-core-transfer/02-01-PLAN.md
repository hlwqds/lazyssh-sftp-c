```go
func (c *SFTPClient) MkdirAll(path string) error {
    c.mu.Lock()
    client := c.client
    c.mu.Unlock()
    if client == nil {
        return fmt.Errorf("not connected: call Connect first")
    }
    // Normalize path
    path = strings.TrimRight(path, "/")
    parts := strings.Split(path, "/")
    for i := 2; i <= len(parts); i++ {
        p := strings.Join(parts[:i], "/")
        if err := client.Mkdir(p); err != nil {
            // Ignore "already exists" errors
            if !strings.Contains(err.Error(), "exists") {
                return fmt.Errorf("sftp mkdir %s: %w", p, err)
            }
        }
    }
    return nil
}
```

`WalkDir` implementation:
```go
func (c *SFTPClient) WalkDir(path string) ([]string, error) {
    c.mu.Lock()
    client := c.client
    c.mu.Unlock()
    if client == nil {
        return nil, fmt.Errorf("not connected: call Connect first")
    }
    var files []string
    err := c.walkDir(client, path, &files)
    return files, err
}

func (c *SFTPClient) walkDir(client *sftp.Client, path string, files *[]string) error {
    entries, err := client.ReadDir(path)
    if err != nil {
        return fmt.Errorf("sftp readdir %s: %w", path, err)
    }
    for _, e := range entries {
        if e.Name() == "." || e.Name() == ".." {
            continue
        }
        fullPath := path + "/" + e.Name()
        if e.IsDir() {
            if err := c.walkDir(client, fullPath, files); err != nil {
                return err
            }
        } else {
            *files = append(*files, fullPath)
        }
    }
    return nil
}
```

Implement `UploadFile`:
1. Open local file with `os.Open(localPath)`, defer close
2. Get file info with `localFile.Stat()` for BytesTotal
3. Call `ts.sftp.CreateRemoteFile(remotePath)`, defer close
4. Custom copy loop with 32KB buffer:
   ```go
   buf := make([]byte, 32*1024)
   var transferred int64
   for {
       n, readErr := src.Read(buf)
       if n > 0 {
           _, writeErr := dst.Write(buf[:n])
           if writeErr != nil {
               return fmt.Errorf("write remote %s: %w", remotePath, writeErr)
           }
           transferred += int64(n)
           if onProgress != nil {
               onProgress(domain.TransferProgress{
                   FileName:   filepath.Base(localPath),
                   FilePath:   localPath,
                   BytesDone:  transferred,
                   BytesTotal: total,
               })
           }
       }
       if readErr != nil {
           if readErr == io.EOF { break }
           return fmt.Errorf("read local %s: %w", localPath, readErr)
       }
   }
   ```
5. Send final Done progress event

Implement `DownloadFile`:
Same pattern but reversed: `sftp.OpenRemoteFile(remotePath)` → `os.Create(localPath)` → copy loop → progress callbacks.

Implement `UploadDir`:
1. First pass: `filepath.WalkDir(localPath, ...)` to count total files (not directories)
2. Call `ts.sftp.MkdirAll(remotePath)` to create root directory
3. Second pass: `filepath.WalkDir(localPath, ...)` for each file:
   - Calculate relative path: `rel, _ := filepath.Rel(localPath, fullPath)`
   - Remote path: `filepath.Join(remotePath, rel)`
   - Create parent dir: `ts.sftp.MkdirAll(filepath.Dir(remoteFile))`
   - Call UploadFile logic (inline, not recursive call to avoid double progress callbacks)
   - Track FileIndex/FileTotal in progress
   - On error: append to failed list, log warning, continue
4. Return failed list (nil if all succeeded)

Implement `DownloadDir`:
Same pattern but reversed: `ts.sftp.WalkDir(remotePath)` → count files → `os.MkdirAll(localPath)` → for each file: calculate relative path, create local dir, download.

Important implementation details:
- Use `filepath.WalkDir` with `fs.DirEntry` for local walks (skip directories)
- Use `filepath.Rel()` to compute relative paths
- On remote: use `/` as separator always (SFTP protocol)
- On local: use `filepath.Join()` for OS-appropriate separator
- For remote path joining in download: use manual string joining with `/`
